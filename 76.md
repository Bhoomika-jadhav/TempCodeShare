class Solution:
    def minWindow(self, s: str, t: str) -> str:
        i,output=0,s
        
        while i<len(s)-1:
            while s[i] not in t:
                i+=1
            tempt = t
            tempstr = s[i]
            tempt = tempt.replace(s[i],'',1)
            j=i+1
            while j<len(s) and len(tempt)>0:
                tempstr+=s[j]
                if s[j] in t:
                    tempt = tempt.replace(s[j],'',1)
                j+=1
            print(tempstr)
            if len(tempt)==0 and len(tempstr)<len(output):
                output = tempstr
            i+=1
        return output
