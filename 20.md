def validParenthesis(s):
    stack = []
    for i in s:
        if i == '(':
            stack += i
        elif i == '[':
            stack += i
        elif i == '{':
            stack += i
        elif i == ')' or i == '}' or i == '}':
            stack.pop()

    if len(stack) == 0:
        return True
    else:
        return False
