class Solution:
    def findMin(self, nums: List[int]) -> int:
        lo, hi = 0, len(nums) - 1

        while lo <= hi:
            if len(nums) == 1: #
                return nums[0]
            
            mid = (lo + hi) // 2
            mid_num = nums[mid]

            if mid >= 0 and mid_num < nums[mid-1]: #
                return mid_num

            elif mid_num < nums[hi]:
                hi = mid-1
            else:
                lo = mid+1

        return -1
